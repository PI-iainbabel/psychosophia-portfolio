Landasan Teori IAT

Implicit Association Test (Greenwald, McGhee, & Schwartz, 1998) mengukur kekuatan asosiasi otomatis antara konsep (misal, gender) dan atribut (misal, karier/keluarga).

Skor D (Greenwald et al., 2003) dihitung dari perbedaan rata-rata RT pada blok gabungan, dibagi standar deviasi gabungan.

Interpretasi: D positif = asosiasi Laki-laki↔Karier, Perempuan↔Keluarga; negatif = sebaliknya.

Referensi Greenwald, A. G., Nosek, B. A., & Banaji, M. R. (2003). Understanding and using the Implicit Association Test: I. An improved scoring algorithm. JPSP, 85(2), 197–216.

Prosedur Eksperimen (7 Blok)

  • Blok 1 – Klasifikasi Gender (Latihan): Kiri = Laki-laki, Kanan = Perempuan
  • Blok 2 – Klasifikasi Atribut (Latihan): Kiri = Karier, Kanan = Keluarga
  • Blok 3 – Gabungan 1 (Latihan): Kiri = Laki-laki+Karier, Kanan = Perempuan+Keluarga
  • Blok 4 – Gabungan 1 (TES): data diambil
  • Blok 5 – Klasifikasi Gender Terbalik (Latihan): Kiri = Perempuan, Kanan = Laki-laki
  • Blok 6 – Gabungan 2 (Latihan): Kiri = Perempuan+Karier, Kanan = Laki-laki+Keluarga
  • Blok 7 – Gabungan 2 (TES): data diambil

Setiap blok diawali instruksi. Partisipan merespons dengan tombol E (kiri) atau I (kanan).

Uji Validitas & Reliabilitas

Validitas Konstruk: Korelasikan skor D dengan skala eksplisit (misal, Ambivalent Sexism). Prediksi r = 0.20–0.40.

Validitas Diskriminan: Bandingkan kelompok yang secara teoretis berbeda (misal, STEM vs Humaniora) dengan uji-t.

Reliabilitas Split-Half: Bagi trial Blok 4 dan 7 menjadi ganjil/genap, hitung D parsial, korelasikan, gunakan Spearman-Brown. Target r > 0.70.

Stabilitas Test-retest: Ulangi setelah 2 minggu, korelasi > 0.60.

Skrip Analisis Data dengan R

Salin dan jalankan skrip berikut di RStudio untuk membersihkan data, menghitung D-score, dan uji hipotesis.

# ================================================================
# SKRIP ANALISIS DATA IAT – GREENWALD ET AL. (2003) PROTOCOL
# ================================================================

# 1. SETUP & BACA DATA
library(dplyr)
library(tidyr)
library(ggplot2)

# Ganti nama file sesuai dengan file CSV yang Anda unduh
# Contoh: "AllData_1782223168009.csv"
data_raw <- read.csv("AllData_1782223168009.csv", header = TRUE, stringsAsFactors = FALSE)

# 2. EKSTRAK DEMOGRAFI (ambil unik per partisipan)
demografi <- data_raw %>%
  group_by(ParticipantID) %>%
  summarise(
    Name = first(Name),
    Age = first(Age),
    Gender = first(Gender),
    Education = first(Education),
    Mode = first(Mode),
    DScore_export = first(DScore),  # D dari aplikasi (sebagai pembanding)
    .groups = "drop"
  )

cat("=== DEMOGRAFI PARTISIPAN ===\n")
print(demografi)

# 3. FILTER RT (hanya RT antara 300 - 10000 ms)
data_valid <- data_raw %>%
  filter(RT_ms >= 300 & RT_ms <= 10000)

cat("\n--- FILTER RT ---\n")
cat("Total trial awal :", nrow(data_raw), "\n")
cat("Total trial valid :", nrow(data_valid), "\n")

# 4. FUNGSI HITUNG D-SCORE PER PARTISIPAN (dengan penalti error)
# Rumus: (Mean Block7 - Mean Block4) / Pooled_SD
# Penalti error: trial error diganti dengan mean blok + 600ms
compute_IAT_D <- function(df_participant) {
  
  # Pisahkan blok 4 dan 7
  b4 <- df_participant %>% filter(Block == 4)
  b7 <- df_participant %>% filter(Block == 7)
  
  # Jika salah satu blok kosong, return NA
  if (nrow(b4) == 0 || nrow(b7) == 0) return(NA)
  
  # Hitung mean awal untuk penalti error (hanya data valid)
  mean_b4 <- mean(b4$RT_ms, na.rm = TRUE)
  mean_b7 <- mean(b7$RT_ms, na.rm = TRUE)
  
  # Terapkan penalti error (IsError == 1)
  b4_adj <- b4 %>%
    mutate(RT_adj = ifelse(IsError == 1, mean_b4 + 600, RT_ms))
  
  b7_adj <- b7 %>%
    mutate(RT_adj = ifelse(IsError == 1, mean_b7 + 600, RT_ms))
  
  # Hitung mean akhir
  mean_final_b4 <- mean(b4_adj$RT_adj, na.rm = TRUE)
  mean_final_b7 <- mean(b7_adj$RT_adj, na.rm = TRUE)
  
  # Hitung standar deviasi masing-masing blok
  sd_b4 <- sd(b4_adj$RT_adj, na.rm = TRUE)
  sd_b7 <- sd(b7_adj$RT_adj, na.rm = TRUE)
  
  # Jika SD = 0 atau NA, return NA
  if (is.na(sd_b4) || is.na(sd_b7) || sd_b4 == 0 || sd_b7 == 0) return(NA)
  
  # Pooled SD (gabungan)
  pooled_sd <- sqrt((sd_b4^2 + sd_b7^2) / 2)
  
  # Skor D
  D <- (mean_final_b7 - mean_final_b4) / pooled_sd
  return(D)
}

# 5. HITUNG D-SCORE UNTUK SEMUA PARTISIPAN
participant_ids <- unique(data_valid$ParticipantID)
hasil_D <- data.frame()

for (pid in participant_ids) {
  df_p <- data_valid %>% filter(ParticipantID == pid)
  d_score <- compute_IAT_D(df_p)
  if (!is.na(d_score)) {
    hasil_D <- rbind(hasil_D, data.frame(ParticipantID = pid, D_score = d_score))
  }
}

# Gabungkan dengan demografi
final_data <- merge(hasil_D, demografi, by = "ParticipantID")

cat("\n=== SKOR D PER PARTISIPAN ===\n")
print(final_data[, c("ParticipantID", "Name", "Age", "Gender", "D_score", "DScore_export")])

# 6. STATISTIK DESKRIPTIF
cat("\n--- DESKRIPTIF SKOR D ---\n")
summary(final_data$D_score)
cat("Standar Deviasi :", sd(final_data$D_score, na.rm = TRUE), "\n")

# 7. UJI HIPOTESIS (t-test berdasarkan Gender)
cat("\n--- UJI-T INDEPENDEN (Laki-laki vs Perempuan) ---\n")
if (length(unique(final_data$Gender)) >= 2) {
  print(t.test(D_score ~ Gender, data = final_data))
} else {
  cat("Hanya ada satu kelompok gender, uji-t tidak dapat dilakukan.\n")
}

# 8. KORELASI DENGAN USIA
cat("\n--- KORELASI DENGAN USIA (Pearson) ---\n")
print(cor.test(final_data$Age, final_data$D_score, method = "pearson"))

# 9. RELIABILITAS SPLIT-HALF (GANJIL-GENAP)
cat("\n--- RELIABILITAS SPLIT-HALF (GANJIL vs GENAP) ---\n")
# Fungsi untuk menghitung D parsial dari subset trial (ganjil/genap)
compute_D_parsial <- function(df_p, parity) {
  # parity: "odd" atau "even"
  df_sub <- df_p %>%
    filter(Block %in% c(4,7)) %>%
    mutate(TrialParity = ifelse(Trial %% 2 == 1, "odd", "even")) %>%
    filter(TrialParity == parity)
  
  if (nrow(df_sub) < 4) return(NA)
  
  b4 <- df_sub %>% filter(Block == 4)
  b7 <- df_sub %>% filter(Block == 7)
  if (nrow(b4) < 2 || nrow(b7) < 2) return(NA)
  
  mean_b4 <- mean(b4$RT_ms, na.rm = TRUE)
  mean_b7 <- mean(b7$RT_ms, na.rm = TRUE)
  sd_b4 <- sd(b4$RT_ms, na.rm = TRUE)
  sd_b7 <- sd(b7$RT_ms, na.rm = TRUE)
  if (is.na(sd_b4) || is.na(sd_b7) || sd_b4 == 0 || sd_b7 == 0) return(NA)
  
  pooled <- sqrt((sd_b4^2 + sd_b7^2) / 2)
  return((mean_b7 - mean_b4) / pooled)
}

# Kumpulkan D parsial untuk setiap partisipan
split_half_df <- data.frame()
for (pid in participant_ids) {
  df_p <- data_valid %>% filter(ParticipantID == pid)
  d_odd <- compute_D_parsial(df_p, "odd")
  d_even <- compute_D_parsial(df_p, "even")
  if (!is.na(d_odd) && !is.na(d_even)) {
    split_half_df <- rbind(split_half_df, 
                           data.frame(ParticipantID = pid, D_odd = d_odd, D_even = d_even))
  }
}

if (nrow(split_half_df) >= 2) {
  korelasi_split <- cor(split_half_df$D_odd, split_half_df$D_even, use = "complete.obs")
  cat("Korelasi Pearson (ganjil vs genap) :", round(korelasi_split, 4), "\n")
  # Koreksi Spearman-Brown
  sb_reliability <- (2 * korelasi_split) / (1 + korelasi_split)
  cat("Reliabilitas Spearman-Brown       :", round(sb_reliability, 4), "\n")
  if (sb_reliability > 0.70) {
    cat("✅ Reliabilitas baik ( > 0.70 )\n")
  } else {
    cat("⚠️ Reliabilitas perlu ditingkatkan\n")
  }
} else {
  cat("⚠️ Data tidak cukup untuk split-half.\n")
}

# 10. VISUALISASI BOXPLOT
cat("\n--- VISUALISASI BOXPLOT ---\n")
p <- ggplot(final_data, aes(x = Gender, y = D_score, fill = Gender)) +
  geom_boxplot(alpha = 0.7, outlier.size = 3) +
  geom_jitter(width = 0.2, alpha = 0.5, size = 2) +
  theme_minimal(base_size = 14) +
  labs(title = "Distribusi Skor D IAT Berdasarkan Gender",
       subtitle = paste("n =", nrow(final_data), "partisipan"),
       y = "Skor D (Greenwald et al., 2003)",
       x = "Jenis Kelamin Partisipan") +
  theme(legend.position = "none",
        plot.title = element_text(face = "bold", hjust = 0.5),
        plot.subtitle = element_text(hjust = 0.5))
print(p)

cat("\n✅ Analisis selesai.\n")
